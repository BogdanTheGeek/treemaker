85/*****
86Add a vertex to the connected component; then add all of its incident
87hinge creases. If the vertex straddles a pseudohinge, its other vertex also
88gets added.
89*****/
90void tmRootNetwork::TryAddVertexToConnectedComponent(tmVertex* aVertex) {
91 TMASSERT(aVertex->mCCFlag != INELIGIBLE)tmOnAssert(!!(aVertex->mCCFlag != INELIGIBLE), L"../Source/tmModel/tmTreeClasses/tmTree_FacetOrder.cpp"
, 91, L"aVertex->mCCFlag != INELIGIBLE");
92 if (aVertex->mCCFlag == ALREADY_ADDED) return;
93 aVertex->mCCFlag = ALREADY_ADDED;
94 mCCVertices.push_back(aVertex);
95 for (size_t i = 0; i < aVertex->mCreases.size(); ++i)
96 TryAddCreaseToConnectedComponent(aVertex->mCreases[i]);
97 tmVertex* mateVertex = aVertex->mLeftPseudohingeMate;
98 if (mateVertex) TryAddVertexToConnectedComponent(mateVertex);
99 mateVertex = aVertex->mRightPseudohingeMate;
100 if (mateVertex) TryAddVertexToConnectedComponent(mateVertex);
101
102 // While normally polys get added to mCCPolys when a crease goes into the
103 // connected component, if the vertex is a leaf node vertex, it's the root
104 // node, and we'll need to add all of its incident polys to our list of polys
105 // because there are no creases extending into the poly from the vertex. We
106 // get the incident polys by looking at the owners of the incident ridge
107 // creases.
108 if (aVertex->mTreeNode->IsLeafNode()) {
109 for (size_t i = 0; i < aVertex->mCreases.size(); ++i) {
110 tmCrease* aCrease = aVertex->mCreases[i];
111 if (!aCrease->IsRidgeCrease()) continue;
112 tmPoly* thePoly = aCrease->GetOwnerAsPoly();
113 TMASSERT(thePoly)tmOnAssert(!!(thePoly), L"../Source/tmModel/tmTreeClasses/tmTree_FacetOrder.cpp"
, 113, L"thePoly");
114 mCCPolys.union_with(thePoly);
115 }
116 }
117}

121Try adding a crease to the connected component. After checking that it's a
122hinge crease, we'll add it; we'll also add its polygon to our polygon list.
123Then we'll add its vertices.
124*****/
125void tmRootNetwork::TryAddCreaseToConnectedComponent(tmCrease* aCrease)

130 aCrease->mCCFlag = ALREADY_ADDED;

, 133, L"thePoly");
134 mCCPolys.union_with(thePoly);
135 TryAddVertexToConnectedComponent(aCrease->mVertices.front());
136 TryAddVertexToConnectedComponent(aCrease->mVertices.back());
137}

159/*****
160Try adding a crease to the spanning tree. Unlike adding to the connected
161component, we'll only add a crease if at least one of its vertices isn't
162already part of the spanning tree.
163*****/
164void tmRootNetwork::TryAddCreaseToSpanningTree(tmCrease* aCrease)
165{
166 if (!aCrease->IsHingeCrease()) return;
167 if (aCrease->mCCFlag == INELIGIBLE) return;
168 if (aCrease->mSTFlag == ALREADY_ADDED) return;
169 tmVertex* v1 = aCrease->mVertices.front();
170 tmVertex* v2 = aCrease->mVertices.back();
171 const bool c1 = (v1->mSTFlag == ALREADY_ADDED);
172 const bool c2 = (v2->mSTFlag == ALREADY_ADDED);
173 if (c1 && c2) return;
174 aCrease->mSTFlag = ALREADY_ADDED;
175 mSTCreases.push_back(aCrease);
176 if (!c1) TryAddVertexToSpanningTree(v1);
177 if (!c2) TryAddVertexToSpanningTree(v2);
178}

191/*****
192Classify the axial vertices by degree within this connected component
193and/or its spanning tree. This populates the arrays mCC0, mCC1, mCC2ST1, mCC2ST2.
194In addition, we check whether any vertex can serve as a connection to another
195polygon cluster and note the possibility in the flag mIsConnectable.
196*****/
197void tmRootNetwork::ClassifyVerticesByDegree()
198{
199 for (size_t i = 0; i < mCCVertices.size(); ++i) {
200 tmVertex* theVertex = mCCVertices[i];
201 if (!theVertex->IsAxialVertex()) continue;
202 size_t ccDegree = theVertex->GetDegree(mCCCreases);
203 size_t stDegree = theVertex->GetDegree(mSTCreases);
204 if (ccDegree == 0) mCC0.push_back(theVertex);
205 else if (ccDegree == 1) mCC1.push_back(theVertex);
206 else if (ccDegree == 2)
207 if (stDegree == 1) mCC2ST1.push_back(theVertex);
208 else if (stDegree == 2) mCC2ST2.push_back(theVertex);
209 else TMFAIL("ClassifyVerticesByDegree(): bad stDegree for ccDegree == 2")tmOnAssert(0, L"../Source/tmModel/tmTreeClasses/tmTree_FacetOrder.cpp"
, 209, L"fail", L"ClassifyVerticesByDegree(): bad stDegree for ccDegree == 2"
);
210 else TMFAIL("ClassifyVerticesByDegree(): bad ccDegree")tmOnAssert(0, L"../Source/tmModel/tmTreeClasses/tmTree_FacetOrder.cpp"
, 210, L"fail", L"ClassifyVerticesByDegree(): bad ccDegree");
211 size_t axDegree = theVertex->GetNumHingeCreases();
212 mIsConnectable |= (axDegree == 2 && ccDegree == 1);
213 }

217/*****
218Add connections to the facet ordering graph (FOG) that splice the pieces
219surrounding this connected component into a single large loop (for depth>0
220pieces) or a sortable graph (for the depth=0 piece). The difference is, in
221the depth=0 piece, we leave out a single splice, which breaks the loop at a
222place that leaves the graph with a single source and sink.
223*****/
224void tmRootNetwork::ConnectFacetGraph()
225{
226 // Vertices of degree 0 in the connected component. There will be either zero
227 // or one of these, and if there is one, that means this constitutes the
228 // entire depth-0 root network.
229 TMASSERT(mCC0.size() <= 1)tmOnAssert(!!(mCC0.size() <= 1), L"../Source/tmModel/tmTreeClasses/tmTree_FacetOrder.cpp"
, 229, L"mCC0.size() <= 1");
230 if (mCC0.size() == 1) {
231 tmVertex* theVertex = mCC0.front();
232 // Unlink across all ridge creases incident on the vertex.
233 for (size_t i = 0; i < theVertex->mCreases.size(); ++i) {
234 tmCrease* theCrease = theVertex->mCreases[i];
235 if (theCrease->IsRidgeCrease()) {
236 tmFacet::Unlink(theCrease->mFwdFacet, theCrease->mBkdFacet);
237 }
238 }
239 // Re-link across all axial creases incident on the vertex (for a border
240 // vertex) or all but one (for an interior vertex).
241 bool needsSkip = !theVertex->IsBorderVertex();
242 for (size_t i = 0; i < theVertex->mCreases.size(); ++i) {
243 tmCrease* theCrease = theVertex->mCreases[i];
244 if (!theCrease->IsBorderCrease() && theCrease->IsAxialCrease()) {
245 if (needsSkip) needsSkip = false;
246 else tmFacet::Link(theCrease->mFwdFacet, theCrease->mBkdFacet);
247 }
248 }
249 // if this connected component had a degree-0 vertex, then it won't have
250 // any more vertices, so we're done.
251 return;
252 }
253
254 // Vertices of degree 2 in the connected component and of degree 2 in the
255 // spanning tree. Swap the links around the vertex.
256 for (size_t i = 0; i < mCC2ST2.size(); ++i) {
257 tmVertex* theVertex = mCC2ST2[i];
258 theVertex->SwapLinks();
259 }
260}

263/*****
264Return true if this network (which is the global root network) can absorb the
265given network (which has a lower depth). This basically boils down to whether
266the given network has a root vertex of degree 1 in the connected component that
267is incident to an axial vertex of the global root network.
268*****/
269bool tmRootNetwork::CanAbsorb(tmRootNetwork* aNetwork, tmVertex*& atVertex)
270{
271 for (size_t i = 0; i < mCCPolys.size(); ++i) {
272 // thePoly is a poly in the global root network
273 tmPoly* thePoly = mCCPolys[i];
274 // theVertex should be one of the non-root axial vertices of thePoly,
275 // i.e., a potential point of attachment of the lower-depth network.
276 // Since we don't have a list specifically of non-root axial vertices,
277 // we'll have to take the long way. We'll go through all paths of the
278 // poly (which own any non-corner axial vertices, which are all we're
279 // interested in); then we'll go through the owned vertices of each path.
280 // We'll still go through a bunch of vertices that have no chance of
281 // being a point of attachment, but that can't be helped.
282 for (size_t j = 0; j < thePoly->mRingPaths.size(); ++j) {
283 tmPath* thePath = thePoly->mRingPaths[j];
284 for (size_t k = 0; k < thePath->mOwnedVertices.size(); ++k) {
285 tmVertex* theVertex = thePath->mOwnedVertices[k];
286 if (theVertex->mDiscreteDepth != aNetwork->mDiscreteDepth) continue;
287 if (!aNetwork->mCC1.contains(theVertex)) continue;
288 atVertex = theVertex;
289 return true;
290 }
291 }
292 }
293 return false;
294}

297/*****
298Absorb the given network, rearranging connections at the given vertex. When we
299absorb a network, we take over its polys (which are used for incidence testing:
300see CanAbsorb()). We don't bother doing anything with the lists of vertices and
301creases because we'll want the global root network to still know its own local
302root vertices and creases.
303*****/
304void tmRootNetwork::Absorb(tmRootNetwork* aNetwork, tmVertex* atVertex)
305{
306 // Swap the links around the vertex being joined.
307 atVertex->SwapLinks();
308
309 // Take over all of the polys of the lower network.
310 mCCPolys.union_with(aNetwork->mCCPolys);
311}

314/*****
315After the global root network has absorbed all other local root networks, we
316break a single link in the large cycle of the facet ordering graph to turn it
317into a proper sortable graph.
318*****/
319void tmRootNetwork::BreakOneLink()
320{
321 // If the root node is a leaf node of the tree, then the original root
322 // network is only a single vertex; in this case, the facet ordering graph is
323 // already sortable and we don't have to do anything.
324 if (mCC0.not_empty()) return;
325
326 // But if the root node was a branch node, then we need to find one link to
327 // break. We'll look through the lists of local root vertices in the global
328 // root network, which are all of depth 0 and thus are the only vertices
329 // eligible for the break.
330
331 // Look first at local root vertices of degree 1 in the connected component.
332 // These will be vertices on the border of the paper. They could be adjacent
333 // to pseudohinges, so we'll have to go to the first non-pseudohinge facets
334 // to either side of the crease to unlink.
335 if (mCC1.not_empty()) {
336 tmVertex* theVertex = mCC1.front();
337 tmCrease* theCrease = theVertex->GetHingeCrease();
338 tmFacet::Unlink(theCrease->GetLeftNonPseudohingeFacet(),
339 theCrease->GetRightNonPseudohingeFacet());
340 return;
341 }
342
343 // If we didn't find one, then we'll have to look at the vertices of degree
344 // 2. There must be at least one vertex of degree 2 in the connected
345 // component and degree 1 in the spanning tree; if they were all of degree 2,
346 // the spanning tree would be a loop. We'll choose one of those to break.
347 TMASSERT(mCC2ST1.not_empty())tmOnAssert(!!(mCC2ST1.not_empty()), L"../Source/tmModel/tmTreeClasses/tmTree_FacetOrder.cpp"
, 347, L"mCC2ST1.not_empty()");
348 tmVertex* theVertex = mCC2ST1.front();
349 tmCrease *crease1, *crease2;
350 theVertex->GetHingeCreases(crease1, crease2);
351 tmFacet::Unlink(crease1->mFwdFacet, crease1->mBkdFacet);
352}

391/*****
392First part of the facet order graph calculation. Construct the array of local
393root networks, which entails calculating the local facet order within each
394poly, identifying the local root vertices and creases, and then building up the
395local root networks, which are the connected components across polys of
396vertices and creases that are local root within their molecule. We then compute
397the spanning tree of each root network and wrap up by classifying all the
398vertices in the network according to their degree in various graphs.
399*****/
400void tmTree::CalcRootNetworks(tmArray<tmRootNetwork*>& rootNetworks)
401{
402 TMASSERT(rootNetworks.empty())tmOnAssert(!!(rootNetworks.empty()), L"../Source/tmModel/tmTreeClasses/tmTree_FacetOrder.cpp"
, 402, L"rootNetworks.empty()");
403
404 // Start by (re)constructing the ordering graph for every poly viewed in
405 // isolation; this will give a collection of disconnected pieces of the graph.
406 // It also constructs the local root vertices and creases for each poly.
407 for (size_t i = 0; i < mOwnedPolys.size(); ++i)
408 mOwnedPolys[i]->CalcLocalFacetOrder();
409
410 // Collect all the local root hinges across all polys and their vertices.
411 tmArray<tmVertex*> localRootVertices;
412 tmArray<tmCrease*> localRootCreases;
413 for (size_t i = 0; i < mOwnedPolys.size(); ++i) {
414 tmPoly* thePoly = mOwnedPolys[i];
